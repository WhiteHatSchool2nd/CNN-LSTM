def normalize(df, TAG_MIN, TAG_MAX):
    ndf = df.copy()
    for c in df.columns:
        if TAG_MIN[c] == TAG_MAX[c]:
            ndf[c] = df[c] - TAG_MIN[c]
        else:
            ndf[c] = (df[c] - TAG_MIN[c]) / (TAG_MAX[c] - TAG_MIN[c])
    return ndf

# Min-Max Normalize
TRAIN_DF = normalize(TRAIN_DF_RAW[VALID_COLUMNS_IN_TRAIN_DATASET], TAG_MIN, TAG_MAX).ewm(alpha=0.9).mean()

window_size = 60
label_size = 80000
def sliding_window_unsupervised(df, window_size, feature_columns, answer_column):
    data = df[feature_columns].values
    answers = answer_column.values

    num_samples = len(df) - window_size
    features = np.empty((num_samples, window_size, len(feature_columns)), dtype=np.float32)
    answer_targets = np.empty(num_samples, dtype=int)

    for i in range(num_samples):
        features[i] = data[i:i+window_size]
        answer_targets[i] = 1 if np.any(answers[i:i+window_size] == 1) else 0

    return features, answer_targets

import torch
import torch.nn as nn

class Autoencoder(nn.Module):
    def __init__(self, n_features, n_hidden, seq_len):
        super(Autoencoder, self).__init__()
        self.seq_len = seq_len
        self.n_hidden = n_hidden

        # CNN 레이어
        self.c1 = nn.Conv1d(in_channels=n_features, out_channels=64, kernel_size=2, stride=1)
        self.relu = nn.ReLU()

        # LSTM 레이어
        self.lstm = nn.LSTM(
            input_size=64,
            hidden_size=n_hidden,
            num_layers=1,
            batch_first=True
        )

        # 선형 레이어
        self.encoder_linear = nn.Linear(in_features=n_hidden, out_features=64)
        self.decoder_linear = nn.Linear(in_features=64, out_features=n_hidden)
        self.lstm_decoder = nn.LSTM(
            input_size=n_hidden,
            hidden_size=n_hidden,
            num_layers=1,
            batch_first=True
        )
        self.output_linear = nn.Linear(in_features=n_hidden, out_features=n_features)

    def forward(self, x):
        x = self.c1(x.permute(0, 2, 1))
        x = self.relu(x)
        x = x.permute(0, 2, 1)

        # LSTM 인코더
        x, (hidden, cell) = self.lstm(x)

        # 인코더 출력
        x = self.encoder_linear(x[:, -1, :])
        x = self.relu(x)
        x = x.unsqueeze(1).repeat(1, self.seq_len, 1)

        # LSTM 디코더
        x, _ = self.lstm_decoder(x, (hidden, cell))

        # 디코더 출력
        x = self.decoder_linear(x)
        x = self.relu(x)
        x = self.output_linear(x)
        return x

def train_model(model, train_loader, val_loader=None, num_epochs=100, verbose=10, patience=10):
    train_hist = []
    val_hist = []
    best_loss = float('inf')
    patience_counter = 0

    for epoch in range(num_epochs):
        model.train()
        train_loss = 0

        for seq, _ in train_loader:
            seq = seq.to(device)
            optimizer.zero_grad()
            output = model(seq)
            loss = loss_fn(output, seq)
            loss.backward()
            optimizer.step()
            train_loss += loss.item()

        train_loss /= len(train_loader)
        train_hist.append(train_loss)

        if val_loader:
            model.eval()
            val_loss = 0
            with torch.no_grad():
                for val_seq, _ in val_loader:
                    val_seq = val_seq.to(device)
                    val_output = model(val_seq)
                    val_step_loss = loss_fn(val_output, val_seq)
                    val_loss += val_step_loss.item()
            val_loss /= len(val_loader)
            val_hist.append(val_loss)

            if val_loss < best_loss:
                best_loss = val_loss
                patience_counter = 0
            else:
                patience_counter += 1

            if patience_counter >= patience:
                print("Early stopping at epoch:", epoch)
                break

        if epoch % verbose == 0:
            print(f'Epoch {epoch}, Train Loss: {train_loss}, Validation Loss: {val_loss}')

    return model, train_hist, val_hist
